591  JnifError::assert(pos == nullptr, "Invalid pos");
592

595  //first = inst;
596  //last = inst;
597  } else {
598  if (pos == nullptr) {
599  p = last;
600  n = nullptr;
601  //last = inst;
602  } else {
603  p = pos->prev;
604  n = pos;
605  }
606  }
607 
608  inst->prev = p;
609  inst->next = n;
610 
611  if (inst->prev != nullptr) {
612  inst->prev->next = inst;
613  } else {
614  first = inst;
615  }
616 
617  if (inst->next != nullptr) {
618  inst->next->prev = inst;
619  } else {
620  last = inst;
621  }
622 
623  _size++;

711  Type TypeFactory::_parseBaseType(const char*& fieldDesc,
712  const char* originalFieldDesc) {
713  switch (*fieldDesc) {
714  case 'Z':
715  return booleanType();
716  case 'B':
717  return byteType();
718  case 'C':
719  return charType();
720  case 'S':
721  return shortType();
722  case 'I':
723  return intType();
724  case 'D':
725  return doubleType();
726  case 'F':
727  return floatType();
728  case 'J':
729  return longType();
730  case 'L': {
731  fieldDesc++;
732 
733  const char* classNameStart = fieldDesc;
734  int len = 0;
735  while (*fieldDesc != ';') {
736  JnifError::check(*fieldDesc != '\0', "");
737  fieldDesc++;
738  len++;
739  }
740 
741  string className(classNameStart, len);
742  return objectType(className);
743  }
