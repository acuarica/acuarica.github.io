71  void parseInstTargets(BufferReader& br, LabelManager& labelManager) {
72  while (!br.eor()) {
73  int offset = br.offset();
74 
75  Opcode opcode = (Opcode) br.readu1();
76  OpKind kind = OPKIND[opcode];
77 
78  switch (kind) {
79  case KIND_ZERO:
80  if (opcode == OPCODE_wide) {
81  u1 opcodew = (Opcode) br.readu1();
82  if (opcodew == OPCODE_iinc) {
83  br.skip(4);
84  } else {
85  br.skip(2);
86  }
87  }
88  break;
89  case KIND_BIPUSH:
90  case KIND_VAR:
91  case KIND_NEWARRAY:
92  br.skip(1);
93  break;
94  case KIND_SIPUSH:
95  case KIND_IINC:
96  case KIND_FIELD:
97  case KIND_INVOKE:
98  case KIND_TYPE:
99  br.skip(2);
100  break;
101  case KIND_MULTIARRAY:
102  br.skip(3);
103  break;
104  case KIND_INVOKEINTERFACE:
105  br.skip(4);
106  break;
107  case KIND_INVOKEDYNAMIC:
108  br.skip(4);
109  break;
110  case KIND_LDC:
111  if (opcode == OPCODE_ldc) {
112  br.readu1();
113  } else {
114  br.readu2();
115  }
116  break;
117  case KIND_JUMP: {
118  short targetOffset = br.readu2();
119 
120  short labelpos = offset + targetOffset;
121  Error::assert(labelpos >= 0,
122  "invalid target for jump: must be >= 0");
123  Error::assert(labelpos < br.size(),
124  "invalid target for jump");
125 
126  labelManager.createLabel(labelpos);
127  break;
128  }
129  case KIND_TABLESWITCH: {
130  for (int i = 0; i < (((-offset - 1) % 4) + 4) % 4; i++) {
131  u1 pad = br.readu1();
132  Error::assert(pad == 0, "Padding must be zero");
133  }
134 
135  {
136  bool check2 = br.offset() % 4 == 0;
137  Error::assert(check2, "%d", br.offset());
138  }
139 
140  int defOffset = br.readu4();
141  labelManager.createLabel(offset + defOffset);
142 
143  int low = br.readu4();
144  int high = br.readu4();
145 
146  Error::assert(low <= high,
147  "low (%d) must be less or equal than high (%d)",
148  low, high);
149 
150  for (int i = 0; i < high - low + 1; i++) {
151  int targetOffset = br.readu4();
152  labelManager.createLabel(offset + targetOffset);
153  }
154  break;
155  }
156  case KIND_LOOKUPSWITCH: {
157  for (int i = 0; i < (((-offset - 1) % 4) + 4) % 4; i++) {
158  u1 pad = br.readu1();
159  Error::assert(pad == 0, "Padding must be zero");
160  }
161 
162  int defOffset = br.readu4();
163  labelManager.createLabel(offset + defOffset);
164 
165  u4 npairs = br.readu4();
166 
167  for (u4 i = 0; i < npairs; i++) {
168  br.readu4(); // Key
169 
170  int targetOffset = br.readu4();
171  labelManager.createLabel(offset + targetOffset);
172  }
173  break;
174  }
175 // case KIND_RESERVED:
176  // break;
177  default:
178  Error::raise("default kind in parseInstTargets: "
179  "opcode: ", opcode, ", kind: ", kind);
180  }
181  }

349  } else if (kind == KIND_RESERVED) {
